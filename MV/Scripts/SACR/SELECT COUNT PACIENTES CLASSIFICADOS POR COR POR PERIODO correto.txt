SELECT A.DH_PRE_ATENDIMENTO_FIM,A.CD_COR_REFERENCIA,
B.CD_ATENDIMENTO,
C.NM_PACIENTE,
E.DS_TIPO_RISCO,
C.NM_BAIRRO,
C.CD_CIDADE


FROM TRIAGEM_ATENDIMENTO A,
DBAMV.ATENDIME B,
DBAMV.PACIENTE C,
DBAMV.SACR_CLASSIFICACAO E


WHERE DH_PRE_ATENDIMENTO_FIM between  to_date('01/03/2016', 'dd/mm/yyyy') and to_date('16/03/2016', 'dd/mm/yyyy')
AND B.CD_ATENDIMENTO = A.CD_ATENDIMENTO
AND C.CD_PACIENTE = B.CD_PACIENTE 
AND E.CD_CLASSIFICACAO = A.CD_CLASSIFICACAO
AND C.CD_CIDADE ='11987'
