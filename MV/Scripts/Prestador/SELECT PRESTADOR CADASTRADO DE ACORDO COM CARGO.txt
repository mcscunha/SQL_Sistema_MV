SELECT  PRESTADOR.NM_PRESTADOR      NM_PRESTADOR,
        PRESTADOR.nr_CPF_CGC        nr_CPF_CGC,
        PRESTADOR.dt_nascimento     dt_nascimento,
        PRESTADOR.nm_mae            nm_mae,
        --PRESTADOR.nm_pai            nm_pai,
        TIP_PRESTA.CD_TIP_PRESTA    CODIGO_TIPO_PRESTADOR,
        TIP_PRESTA.NM_TIP_PRESTA    DESCRICAO_TIPO_PRESTADOR
  FROM DBAMV.PRESTADOR                  PRESTADOR,
     --  DBAMV.ESPECIALID                 ESPECIALID,
       --DBAMV.ESP_MED                    ESP_MED,
       DBAMV.TIP_PRESTA                 TIP_PRESTA
 WHERE --PRESTADOR.CD_PRESTADOR   = ESP_MED.CD_PRESTADOR
  -- AND ESP_MED.CD_ESPECIALID    = ESPECIALID.CD_ESPECIALID
   PRESTADOR.CD_TIP_PRESTA  = TIP_PRESTA.CD_TIP_PRESTA
   and TIP_PRESTA.CD_TIP_PRESTA = 8
   order by TIP_PRESTA.NM_TIP_PRESTA, PRESTADOR.NM_PRESTADOR
