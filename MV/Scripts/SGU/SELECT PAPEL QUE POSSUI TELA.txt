SELECT
PAPEL_MOD.CD_PAPEL,
--PAPEL.ds_papel,
--MOD_SIS.CD_SISTEMA
MODULOS.CD_MODULO
--MODULOS.TP_MODULO
FROM DBASGU.MOD_SIS MOD_SIS, DBASGU.MODULOS MODULOS, DBASGU.PAPEL_MOD PAPEL_MOD, DBASGU.PAPEL PAPEL
    WHERE 
    MOD_SIS.CD_MODULO = MODULOS.CD_MODULO 
    AND
    PAPEL_MOD.cd_modulo = MOD_SIS.cd_modulo   
      AND MODULOS.CD_MODULO = 'M_PREMED'
      GROUP BY PAPEL_MOD.CD_PAPEL, /*,PAPEL.ds_papel*/ /*--,MOD_SIS.CD_SISTEMA, */MODULOS.CD_MODULO/*, MODULOS.TP_MODULO*/
      ORDER BY CD_PAPEL
