 SELECT COUNT (DBAMV.EXA_LAB.NM_EXA_LAB)
        ,DBAMV.EXA_LAB.NM_EXA_LAB
        , DBAMV.PED_LAB.NM_PRESTADOR
        
 FROM DBAMV.PED_LAB
      ,DBAMV.ITPED_LAB
      ,DBAMV.EXA_LAB  
 WHERE (DBAMV.PED_LAB.CD_PED_LAB = DBAMV.ITPED_LAB.CD_PED_LAB
    AND DBAMV.EXA_LAB.CD_EXA_LAB = DBAMV.ITPED_LAB.CD_EXA_LAB
       )
    AND DBAMV.PED_LAB.DT_PEDIDO >= '20/09/2011' 
    AND NM_PRESTADOR IS NOT NULL
    AND DBAMV.PED_LAB.CD_CONVENIO = 2 --1=INTERNACAO 2=AMBULATORIO
    AND (DBAMV.EXA_LAB.CD_PROCEDIMENTO_SIA IN () OR DBAMV.EXA_LAB.CD_PROCEDIMENTO_SIH IN ())
 GROUP BY  DBAMV.EXA_LAB.NM_EXA_LAB, DBAMV.PED_LAB.NM_PRESTADOR     
 
 
 ORDER BY 1 DESC, 2
