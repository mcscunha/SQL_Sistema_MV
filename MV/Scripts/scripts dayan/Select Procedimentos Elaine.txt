SELECT DBAMV.Procedimento_Sus_Valor.CD_PROCEDIMENTO, 
DS_PROCEDIMENTO,DBAMV.Procedimento_Sus_Valor.DT_VIGENCIA, 
--vl_servico_ambulatorial, 
--VL_SERVICO_HOSPITALAR, 
--VL_SERVICO_PROFISSIONAL, 
VL_TOTAL_INTERNACAO, 
--VL_SERVICO_PROFISSIONAL_AMB, 
VL_TOTAL_AMBULATORIAL   
FROM DBAMV.PROCEDIMENTO_SUS, DBAMV.Procedimento_Sus_Valor
where DBAMV.PROCEDIMENTO_SUS.CD_PROCEDIMENTO =  DBAMV.Procedimento_Sus_Valor.CD_PROCEDIMENTO
--AND CD_SUB_GRUPO_PROCEDIMENTO = 2
AND (DS_PROCEDIMENTO LIKE '%DOPPLER%' OR DS_PROCEDIMENTO LIKE '%CATETERISMO%' OR DS_PROCEDIMENTO LIKE '%TOMOGRA%' OR DS_PROCEDIMENTO LIKE '%ECOCARDIOGRAMA%' 
OR DS_PROCEDIMENTO LIKE '%CAROTIDA%')
AND (VL_TOTAL_INTERNACAO <> 0 OR VL_TOTAL_AMBULATORIAL <> 0)
AND DT_VIGENCIA >= '01/01/2011'
ORDER BY DBAMV.Procedimento_Sus.DS_PROCEDIMENTO, DBAMV.Procedimento_Sus_Valor.DT_VIGENCIA DESC
