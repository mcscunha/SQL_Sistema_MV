SELECT PROCEDIMENTO_SUS.CD_PROCEDIMENTO,
       PROCEDIMENTO_SUS.DS_PROCEDIMENTO,
       DBAMV.PROCEDIMENTO_CID_VIGENCIA.CD_CID,
       CID.DS_CID
  FROM DBAMV.PROCEDIMENTO_SUS PROCEDIMENTO_SUS
 INNER JOIN DBAMV.PROCEDIMENTO_CID_VIGENCIA
    ON DBAMV.PROCEDIMENTO_CID_VIGENCIA.CD_PROCEDIMENTO =
       DBAMV.PROCEDIMENTO_SUS.CD_PROCEDIMENTO
 INNER JOIN CID
    ON DBAMV.PROCEDIMENTO_CID_VIGENCIA.CD_CID = CID.CD_CID
 WHERE PROCEDIMENTO_SUS.CD_PROCEDIMENTO = 0303170158
